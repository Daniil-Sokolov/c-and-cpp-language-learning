``` c语言
	order(a[10]);

		printf("%d ", a[i]);
	}

int order(int a[10] ){
	int max, i, j;
	max = a[9];
	for(i = 9; i > -1; i--){
		max = a[i];
		for (j = 0; j < i; j++) {
			if(max < a[j]) {
				max = a[j];
				a[j] = a[i];
				a[i] = max;
			}
		}
	}
	return a[10];
}
```
