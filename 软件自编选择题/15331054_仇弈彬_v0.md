```
#include<stdio.h>
int Max(int,int);
int main(){
	int num1, num2, num;
	
	scanf("%d%d",&num1,&num2);
	num = Max(num1,num2);
	printf("%d\n",m);
	
	return 0;
}
int Max(int a,int b){
	int max;
	
	if(a >= b) {
		max = a;
	}
	else {
		max = b;
	}
	
	return max;
}
```
A.num1,num2; a,b;
B.a,b; num1,num2;
C.a,b; max,num;
D.a,num1; b,num2;

Answer:B
The definition of the formal argument and actual argument.

2.What is the output of the following program?
```
#include<stdio.h>
int main(){
	char a[3][3] = {"as",'*','=','\0'};
	printf("!%c!\n",a[0][2]);
	printf("!%c!\n",a[1][0]);
	return 0;
} 
```

！ ！
！*！ 

B.

！s！
！*！

C.

Compile Error

Answer:A

```
	int num3 = 23;

int f(int a,int b,int c){
	int s = 100;
	
	if(a + b > c && abs(a - b) < c){
		if(pow(a,2) + pow(b,2) == pow(c,2)

		|| pow(b,2) + pow(c,2) == pow(a,2)){
			printf("%d\n",s);
		}
		else{
			printf("%d\n",2 * s);
		}
	}
	else{
		printf("Error\n");
	}
	return 0;
}
```
